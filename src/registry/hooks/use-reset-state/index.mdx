## API

```ts
type ResetState = () => void

/**
 * A hook to reset a state
 * @param initialState - The initial state
 */
export function useResetState<S>(
  initialState: S | (() => S),
): readonly [S, Dispatch<SetStateAction<S>>, ResetState]
```
